CREATE DATABASE SISTEMA_DE_MONITORAMENTO_DE_RUIDO;

USE SISTEMA_DE_MONITORAMENTO_DE_RUIDO;

CREATE TABLE MEDIDAS(
    ID_DO_SENSOR INT NOT NULL,
    ORDEM INT NOT NULL AUTO_INCREMENT,
    DATA_E_HORA DATETIME NOT NULL,
    NIVEL_DE_RUIDO DECIMAL(5,2) NOT NULL,
    PRIMARY KEY(ID_DO_SENSOR, ORDEM)
)ENGINE=MyISAM;
